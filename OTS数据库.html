        // 显示职业详情

        function showOccupationDetails(code) {

            const occupation = occupationData.find(item => item.code === code);

            

            if (!occupation) {

                return;

            }

            

            // 更新UI

            occupationName.textContent = occupation.occupation;

            occupationCode.textContent = occupation.code;

            occupationDescription.textContent = occupation.task_chinese || "暂无职业介绍信息";

            

            // 处理工作任务，将<br>转换为HTML

            if (occupation.task && occupation.task.trim() !== "") {

                // 分割任务项

                const taskItems = occupation.task.split('<br>').filter(item => item.trim() !== "");

                

                // 清空现有内容

                occupationTasks.innerHTML = '';

                

                // 添加每个任务项

                taskItems.forEach(task => {

                    const taskElement = document.createElement('div');

                    taskElement.className = 'task-item';

                    

                    // 移除可能的编号前缀（如"1. "）

                    const cleanTask = task.replace(/^\d+\.\s*/, '');

                    taskElement.textContent = cleanTask;

                    

                    occupationTasks.appendChild(taskElement);

                });

            } else {

                occupationTasks.innerHTML = '<p>暂无工作任务信息</p>';

            }

            

            // 显示详情并隐藏空消息

            emptyMessage.style.display = 'none';

            occupationDetails.style.display = 'block';

            resultContainer.style.display = 'block';

            

            // 滚动到结果区域

            resultContainer.scrollIntoView({ behavior: 'smooth', block: 'start' });

        }


        // 事件监听器

        occupationSelect.addEventListener('change', function() {

            if (this.value) {

                showOccupationDetails(this.value);

            } else {

                // 显示空消息

                emptyMessage.style.display = 'block';

                occupationDetails.style.display = 'none';

                resultContainer.style.display = 'block';

            }

        });


        // 页面加载时初始化

        document.addEventListener('DOMContentLoaded', function() {

            initializeSelect();

            

            // 如果URL中有职业代码参数，自动选择

            const urlParams = new URLSearchParams(window.location.search);

            const codeParam = urlParams.get('code');

            if (codeParam) {

                const occupation = occupationData.find(item => item.code === codeParam);

                if (occupation) {

                    occupationSelect.value = codeParam;

                    showOccupationDetails(codeParam);

                }

            }

            

            // 添加一些示例数据到控制台，方便调试

            console.log(`已加载 ${occupationData.length} 个职业数据`);

        });

    </script>
